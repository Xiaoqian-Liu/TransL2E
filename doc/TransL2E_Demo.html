# a function to simulate the covariance matrix
source_cov = function(p, rho){
  
  source_covariance = matrix(0, nrow = p, ncol = p)
  for (i in 1:p) {
    for (j in 1:p) {
      source_covariance[i,j] = rho^abs(i-j)
    }
  }
  return(source_covariance)
}

# a function to create given number of source datasets
source_cre <- function(n_s, p, p_s, beta_t, snr, sigma_s, Sigma_S, K, out_prop, out_sig) {
  
  # Generate a list of K source datasets
  source_list <- lapply(seq_len(K), function(i) {
    # Generate design matrix for the current source
    X <- mvrnorm(n = n_s, mu = rep(0, p), Sigma = Sigma_S)
    
    # Create a shift vector for beta
    delta_s <- rep(0, p)
    shift_index = sample(1:p, p_s, replace = F)
    delta_s[shift_index] <- rnorm(p_s, mean = 0, sd = sigma_s)
    #delta_s[1:p_s] <- rnorm(p_s, mean = 0, sd = sigma_s)
    
    # Adjust beta_t with the current shift to obtain beta_s
    beta_s <- beta_t + delta_s
    
    # Generate response vector Y with additive noise
    Y <- X %*% beta_s + rnorm(n_s, mean = 0, sd = snr)
    
    # Introduce outliers to a proportion of the responses
    n_out <- floor(n_s * out_prop)
    Y[1:n_out] <- Y[1:n_out] + out_sig*max(abs(Y))
    
    # Introduce outliers to a proportion of the covariates
    #X[1:n_out, ] = X[1:n_out, ] + out_sig
    
    # Return the source dataset in the desired list format
    list(x = X, y = Y)
  })
  
  return(source_list)
}


# define parameters
sigma_snr = 1

p = 100 # parameters
n_t = 200 # target sample size
n_s = 400 # source sample size

K = 5 # number of source

rho = 0.5 # level of covariate shift
Sigma = source_cov(p, rho)

beta_t = matrix(0, nrow = p)
beta_t[1:(0.1*p), 1] = 1 # 10% sparsity

sigma_ms = 0.2 # model shift level
p_s = 0.3*p

# outlier proportion and signal level
out_prop = 0.1
out_sig = 2

# lambda seuence
lambda_seq = exp(seq(log(10), log(10^-4), length.out = 20))

set.seed(1)
# simulate the target sample
X_t = matrix(rnorm(n_t * p), nrow = n_t, ncol = p) # target covariates
y_t = X_t %*% beta_t + rnorm(n_t, mean = 0, sd = sigma_snr) # target response
    
# add outler to the samples
y_t[1:(n_t * out_prop)] = y_t[1:(n_t * out_prop)] + out_sig*max(y_t)

# simulate the source dataset
cur_data = source_cre(n_s = n_s, p = p, p_s = p_s, beta_t = beta_t,
                      snr = sigma_snr, sigma_s = sigma_ms, Sigma_S = Sigma,
                      K = K, out_prop = out_prop, out_sig = out_sig)

set.seed(1)
# Run Base L2E on Target Data Only
Time1_start = proc.time()
    
# Fit the l2e lasso model
sol_mm_cv <- CV_L2E_sparse_ncv(y_t, X_t, penalty = 'lasso', nfolds = 5, trace = FALSE)

      
sol_mm_opt = L2E_sparse_ncv(y_t, X_t, lambdaSeq = sol_mm_cv$lambda.1se, penalty = 'lasso', Show.Time = FALSE)
      
Time_plain_l2e = (proc.time() - Time1_start)[[3]]

# square error for parameter estimation
plain_l2e = norm(beta_t - sol_mm_opt$Beta, '2')/norm(beta_t, '2')
        
TP_plain_l2e <- length(intersect(which(abs(sol_mm_opt$Beta)>=0.05), which(beta_t!=0)))
FP_plain_l2e <- length(intersect(which(abs(sol_mm_opt$Beta)>=0.05), which(beta_t==0)))
FN_plain_l2e <- length(intersect(which(abs(sol_mm_opt$Beta)<=0.05), which(beta_t!=0)))
F1_plain_l2e <- 2*TP_plain_l2e/(2*TP_plain_l2e+FP_plain_l2e+FN_plain_l2e)
SR_plain_l2e <- (TP_plain_l2e+(length(beta_t)-TP_plain_l2e-FP_plain_l2e-FN_plain_l2e))/length(beta_t)
TPR_plain_l2e <- TP_plain_l2e/length(which(beta_t!=0))
FPR_plain_l2e <- FP_plain_l2e/length(which(beta_t==0))
        

print(paste0('Base L2E Estimation Error: ', round(plain_l2e, 3)))
